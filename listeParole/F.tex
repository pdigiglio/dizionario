
\newglossaryentry{florilegio}{% 10
    name        = {Florilegio},%
    description = {\switch{1}~\letter~Raccolta di brani tratti dall'opera di uno o più scrittori, antologia: \ex{florilegio carducciano}. \switch{2}~\eccl~Raccolta contenente preghiere e vite di santi. \switch{3}~Raccolta di brutture, modelli da non imitare: \ex{una prosa che è un florilegio di trivialità}}%
}
